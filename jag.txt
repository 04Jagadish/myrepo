import java.lang.Math;

class Main {
  public static void main(String[] args) {

    // create variable in Degree
    double a = 30;
    double b = 45;

    // convert to radians
    a = Math.toRadians(a);
    b = Math.toRadians(b);

    // print the cosine value
    System.out.println(Math.cos(a));  // 0.8660254037844387
    System.out.println(Math.cos(b));  // 0.7071067811865476
  }
}
